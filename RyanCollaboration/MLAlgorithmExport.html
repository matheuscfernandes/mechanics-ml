def ITV(table,time): #IndexOfTValue
    found=False; index=-1
    while not found:
        index+=1
        if table[index,0]>=time:
            found=True
    return index

def NormalizeTime(tableAll):
    tableAllNew=[]
    for table in tableAll:
        minTime=min(table[:,0])
        table[:,0]=table[:,0]-minTime
        tableAllNew.append(table)
    return tableAllNew

def ConvertEventsToPoints(EventsAll):
    Points=[]
    for events in EventsAll:
        for P in events:
            Points.append(P)
    return Points
    
def ConvertEventsYToPoints(EventsAll,EventsYAll):
    Points=[]
    ct=-1
    for events in EventsAll:
        ct+=1
        for P in events:
            Points.append(EventsYAll[ct])
    return Points
    
def CreateEventsTableViaP(table,pLowerThreshold,nPoints):
    P=table[:,1]
    #finding index for cutting table
    BI=[] #BeginingIndex
    EI=[] #EndingIndex
    B=True #Beginning Trigger
    E=False #Ending Trigger
    
    for i in range(nPoints,len(P)-nPoints):
        if (P[i+nPoints]-P[i-nPoints])>0 and P[i]>=pLowerThreshold and B==True:
            BI.append(i)
            B=False
            E=True
        elif (P[i+nPoints]-P[i-nPoints])<0 and E==True:
            EI.append(i)
            B=True
            E=False
        elif i == (len(P)-nPoints-1) and E==True:
            EI.append(i)
        else: pass
        
    table_new=[]
    for c in range(len(BI)):
        table_temp=np.copy(table[BI[c]:EI[c],:])
        table_new.append(table_temp)
        
    return table_new

def KeyToName(table,table_names):
    return_table=[]
    for i in table:
        for j in range(len(table_names)):
            if i == j: return_table.append(table_names[j])
            else: pass
    return return_table

def InterpolateDF(DF):
    Time=DF.index
    InterpolatedDF=DF.copy(deep=True)
    
    for Name in ['Contact','Curvature','Inflation']:
        col=DF[Name]
        missingVals=pd.isnull(col)
        
        Interpolated=scipy.interpolate.interp1d(Time[np.invert(missingVals)],col[np.invert(missingVals)],fill_value="extrapolate",kind='cubic')
        InterpolatedDF[Name]=Interpolated(list(Time))
        
    return InterpolatedDF
